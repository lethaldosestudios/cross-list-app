        </div>

            <button className="neumorph-button p-2 opacity-0 group-hover:opacity-100 transition-opacity">

            </button>
          </DropdownMenuTrigger>
          <DropdownMenuContent className="neumorph-card border-none">
            <DropdownMenuItem onClick={() => onEdit(product)}>
